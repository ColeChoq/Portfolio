           while ( shared_ring.spaces [j] == 0 ) {
                pthread_cond_wait ( &prod_cond [j], &prod [j] );
          }



    	  shared_ring.flavor[j][shared_ring.in_ptr[j]] = shared_ring.serial[j];//put donut in queue

	  shared_ring.serial[j] = shared_ring.serial[j] + 1;//increment serial number
	  shared_ring.in_ptr[j] = (shared_ring.in_ptr[j] + 1) % NUMSLOTS; //increment in ptr

	  shared_ring.donuts[j] = shared_ring.donuts[j] + 1; //increment donut count

	}

int first_fit(struct request* request) {

    struct free_list* fList = NULL;

    // looking for first available block - look through all elements
    for (fList = list_head.next; fList; fList = fList->next) {
        if (request->size <= fList->block_size) {

            request->is_allocated = TRUE;
            request->base_adr = fList->block_adr;
            request->next_boundary_adr = request->base_adr + request->size;

            total_free = total_free - request->size;
            request->memory_left = total_free;


            if ((fList->block_size = fList->block_size - request->size) == 0) {     // Make sure they arent the same size
                fList->previous->next = fList->next;
                fList->next->previous = fList->previous;

                free(fList);
                request->elements_on_free_list = getCount(list_head.next);
                return 0;
            }

            fList->block_adr = fList->block_adr + request->size;
            request->elements_on_free_list = getCount(list_head.next);
            return 0;
        }
    }

    // No block big enough
    request->memory_left = total_free;//only need to update this as all others are preset to fail
    return 0;
}


int best_fit(struct request* request) {

    struct free_list* current = NULL, * smBlock = NULL;      // free list / smallest block
    int difference = 0, remainder = 0, start = TRUE;

    // Find the smallest block that fits
    for (current = list_head.next; current; current = current->next) {
        if (request->size <= current->block_size) {

            difference = current->block_size - request->size;

            if (remainder >= difference) {   //if Difference is less than the remainder we know this is a better fit
                remainder = difference;
                smBlock = current;
            }

            // Only used for the start of the list
            if (start) {
                remainder = difference;
            }

            start = FALSE;    // No longer first itteration
        }
    }
    if (list_head.next->next == NULL) {
        smBlock = list_head.next;
    }
    if (smBlock != NULL) {      //Verify it's not null and set the block
        request->is_allocated = TRUE;
        request->base_adr = smBlock->block_adr;
        request->next_boundary_adr = request->base_adr + request->size;

        total_free = total_free - request->size;
        request->memory_left = total_free;

        // combine blocks if needed
        if ((smBlock->block_size = smBlock->block_size - request->size) == 0) {
            smBlock->previous->next = smBlock->next;
            smBlock->next->previous = smBlock->previous;

            if ((smBlock->block_size = smBlock->previous->block_adr + smBlock->previous->block_size) == 0) {
                smBlock->block_adr = smBlock->previous->block_adr;
                smBlock->block_size = smBlock->block_size + smBlock->previous->block_size;
            }
            free(smBlock);
            request->elements_on_free_list = getCount(list_head.next);
            return 0;
        }
        if ((smBlock->block_size = smBlock->previous->block_adr + smBlock->previous->block_size) == 0) {
            smBlock->block_adr = smBlock->previous->block_adr;
            smBlock->block_size = smBlock->block_size + smBlock->previous->block_size;
            free(smBlock);
            request->elements_on_free_list = getCount(list_head.next);
            return 0;
        }

        smBlock->block_adr = smBlock->block_adr + request->size;
        request->elements_on_free_list = getCount(list_head.next);
        return 0;
    }

    // No Blocks large enough
    request->memory_left = total_free;
    request->elements_on_free_list = getCount(list_head.next);
    return 0;
}



int update_list(int elemt, int rem) {
    struct free_list* current, * new_block, * temp;

    if (req_array[elemt].is_allocated == FALSE) { //If it wan't allocated return immediately
        return 0;
    }

    req_array[elemt].is_allocated = DONE;       //tells us if it has been allocated then freed
    total_free += req_array[elemt].size;        //increases the size of freed memory 

    // increment through the list
    for (current = list_head.next; current; current = current->next) {


        if (req_array[elemt].base_adr > current->block_adr) {       //if it is beyond check later
            continue;
        }

        // new block creation
        new_block = malloc(sizeof(struct free_list));
        new_block->block_adr = req_array[elemt].base_adr;
        new_block->block_size = req_array[elemt].size;
        new_block->adjacent_adr = new_block->block_adr + new_block->block_size;
        new_block->next = current;
        current->previous->next = new_block;
        new_block->previous = current->previous;
        current->previous = new_block;

        //can combine with next ?
        if (new_block->adjacent_adr == new_block->next->block_adr) {
            temp = new_block->next;
            new_block->block_size = new_block->block_size + new_block->next->block_size;
            new_block->adjacent_adr = new_block->next->adjacent_adr;
            new_block->next = new_block->next->next;

            if (new_block->next) {
                new_block->next->previous = new_block;
            }

            free(temp);
        }

        // can we combine with previous ?
        current = new_block->previous;

        if ((current != NULL) && (current->adjacent_adr != 0)) {

            if (current->adjacent_adr == current->next->block_adr) {
                temp = current->next;
                current->block_size = current->block_size + current->next->block_size;
                current->adjacent_adr = current->next->adjacent_adr;
                current->next = current->next->next;

                if (current->next) {
                    current->next->previous = current;
                }

                free(temp);
            }
        }
        req_array[rem].elements_on_free_list = getCount(list_head.next);
        return 0;
    }
}
